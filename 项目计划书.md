### 4.3 数据库设计（核心表）

```sql
-- 用户表
CREATE TABLE `user` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `username` VARCHAR(50) UNIQUE NOT NULL COMMENT '用户名',
    `password` VARCHAR(100) NOT NULL COMMENT '密码',
    `nickname` VARCHAR(50) COMMENT '昵称',
    `avatar` VARCHAR(255) COMMENT '头像URL',
    `signature` VARCHAR(255) COMMENT '个性签名',
    `status` TINYINT DEFAULT 1 COMMENT '状态 1-在线 2-离线 3-忙碌 4-勿扰',
    `last_login_time` DATETIME COMMENT '最后登录时间',
    `last_login_ip` VARCHAR(50) COMMENT '最后登录IP',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX `idx_username` (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- 好友关系表
CREATE TABLE `friend` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `user_id` BIGINT NOT NULL COMMENT '用户ID',
    `friend_id` BIGINT NOT NULL COMMENT '好友ID',
    `remark` VARCHAR(50) COMMENT '好友备注',
    `group_name` VARCHAR(50) DEFAULT '默认分组' COMMENT '分组名称',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY `uk_friend` (`user_id`, `friend_id`),
    INDEX `idx_friend_id` (`friend_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='好友关系表';

-- 好友申请表
CREATE TABLE `friend_apply` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `applicant_id` BIGINT NOT NULL COMMENT '申请人ID',
    `target_id` BIGINT NOT NULL COMMENT '目标用户ID',
    `message` VARCHAR(255) COMMENT '申请消息',
    `status` TINYINT DEFAULT 0 COMMENT '状态 0-待处理 1-已接受 2-已拒绝 3-已忽略',
    `reject_reason` VARCHAR(255) COMMENT '拒绝理由',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX `idx_target` (`target_id`, `status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='好友申请表';

-- 消息表
CREATE TABLE `message` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `msg_id` VARCHAR(64) UNIQUE NOT NULL COMMENT '消息唯一ID（用于幂等）',
    `sender_id` BIGINT NOT NULL COMMENT '发送者ID',
    `receiver_id` BIGINT NOT NULL COMMENT '接收者ID（用户ID或群组ID）',
    `chat_type` TINYINT NOT NULL COMMENT '聊天类型 1-私聊 2-群聊',
    `msg_type` TINYINT NOT NULL COMMENT '消息类型 1-文本 2-图片 3-语音 4-文件 5-系统',
    `content` TEXT COMMENT '消息内容',
    `media_url` VARCHAR(500) COMMENT '媒体文件URL',
    `thumbnail_url` VARCHAR(500) COMMENT '缩略图URL',
    `reply_msg_id` VARCHAR(64) COMMENT '引用消息ID',
    `at_user_ids` VARCHAR(500) COMMENT '@用户ID列表',
    `is_recalled` TINYINT DEFAULT 0 COMMENT '是否撤回 0-否 1-是',
    `is_read` TINYINT DEFAULT 0 COMMENT '是否已读 0-否 1-是',
    `expire_time` DATETIME COMMENT '阅后即焚过期时间',
    `seq_id` BIGINT COMMENT '消息序列号（用于排序）',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX `idx_receiver` (`receiver_id`, `create_time`),
    INDEX `idx_sender` (`sender_id`, `create_time`),
    INDEX `idx_msg_id` (`msg_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='消息表';

-- 会话表
CREATE TABLE `session` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `user_id` BIGINT NOT NULL COMMENT '用户ID',
    `target_id` BIGINT NOT NULL COMMENT '目标ID（用户ID或群组ID）',
    `chat_type` TINYINT NOT NULL COMMENT '聊天类型 1-私聊 2-群聊',
    `last_msg_id` BIGINT COMMENT '最后一条消息ID',
    `last_msg_content` VARCHAR(255) COMMENT '最后消息内容预览',
    `last_msg_time` DATETIME COMMENT '最后消息时间',
    `unread_count` INT DEFAULT 0 COMMENT '未读消息数',
    `is_pinned` TINYINT DEFAULT 0 COMMENT '是否置顶 0-否 1-是',
    `is_muted` TINYINT DEFAULT 0 COMMENT '是否免打扰 0-否 1-是',
    `is_archived` TINYINT DEFAULT 0 COMMENT '是否归档 0-否 1-是',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    UNIQUE KEY `uk_user_target` (`user_id`, `target_id`, `chat_type`),
    INDEX `idx_user` (`user_id`, `is_archived`, `last_msg_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='会话表';

-- 群组表
CREATE TABLE `group_info` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `group_name` VARCHAR(100) NOT NULL COMMENT '群名称',
    `group_avatar` VARCHAR(255) COMMENT '群头像',
    `group_notice` TEXT COMMENT '群公告',
    `owner_id` BIGINT NOT NULL COMMENT '群主ID',
    `member_count` INT DEFAULT 0 COMMENT '成员数量',
    `max_member_count` INT DEFAULT 500 COMMENT '最大成员数',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='群组表';

-- 群成员表
CREATE TABLE `group_member` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `group_id` BIGINT NOT NULL COMMENT '群组ID',
    `user_id` BIGINT NOT NULL COMMENT '用户ID',
    `nickname` VARCHAR(50) COMMENT '群昵称',
    `role` TINYINT DEFAULT 0 COMMENT '角色 0-普通成员 1-管理员 2-群主',
    `join_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY `uk_group_user` (`group_id`, `user_id`),
    INDEX `idx_user` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='群成员表';

-- 扫码登录表
CREATE TABLE `qrcode_login` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `token` VARCHAR(64) UNIQUE NOT NULL COMMENT '二维码token',
    `status` TINYINT DEFAULT 0 COMMENT '状态 0-待扫码 1-已扫码待确认 2-已确认 3-已过期',
    `user_id` BIGINT COMMENT '扫码用户ID',
    `device_info` VARCHAR(255) COMMENT '设备信息',
    `expire_time` DATETIME NOT NULL COMMENT '过期时间',
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='扫码登录表';

-- 用户设备表（多端管理）
CREATE TABLE `user_device` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `user_id` BIGINT NOT NULL COMMENT '用户ID',
    `device_id` VARCHAR(100) NOT NULL COMMENT '设备ID',
    `device_name` VARCHAR(100) COMMENT '设备名称',
    `device_type` TINYINT COMMENT '设备类型 1-Android 2-iOS 3-Web',
    `login_time` DATETIME COMMENT '登录时间',
    `last_active_time` DATETIME COMMENT '最后活跃时间',
    `is_online` TINYINT DEFAULT 0 COMMENT '是否在线',
    INDEX `idx_user` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户设备表';
```

### 4.4 缓存策略设计

| 缓存对象 | 缓存Key | 过期时间 | 更新策略 |
|----------|---------|----------|----------|
| 用户信息 | `user:info:{userId}` | 30分钟 | 写时更新，LRU淘汰 |
| 好友列表 | `friend:list:{userId}` | 10分钟 | 增删时失效 |
| 会话列表 | `session:list:{userId}` | 5分钟 | 新消息时更新 |
| 在线状态 | `user:online:{userId}` | 5分钟（续期） | 心跳续期 |
| 登录Token | `token:{userId}` | 7天 | 登录时创建 |
| 扫码状态 | `qrcode:{token}` | 5分钟 | 状态变更时更新 |
| 离线消息队列 | `offline:msg:{userId}` | 7天 | 上线消费后删除 |

### 4.6 WebSocket 重连机制

```
连接断开
    │
    ▼
等待 1秒 ──→ 重连失败 ──→ 等待 2秒 ──→ 重连失败 ──→ 等待 4秒
    │                                              │
    ▼                                              ▼
重连成功 ──→ 拉取离线消息                           重连失败
    │                                              │
    ▼                                              ▼
恢复正常                                    等待 8秒（最大间隔）
                                                   │
                                                   ▼
                                            继续重试（最多10次）
                                                   │
                                                   ▼
                                            提示用户手动重连
```

### 4.7 接口限流策略

| 接口类型 | 限流规则 | 说明 |
|----------|----------|------|
| 登录接口 | 5次/分钟/IP | 防止暴力登录 |
| 注册接口 | 3次/分钟/IP | 防止恶意注册 |
| 搜索接口 | 10次/分钟/用户 | 防止频繁搜索 |
| 发送消息 | 60次/分钟/用户 | 正常使用不限制 |
| 文件上传 | 10次/分钟/用户 | 防止滥用存储 |
| 全局默认 | 100次/分钟/用户 | 兜底限流 |

## 5. 风险管理

### 5.1 技术风险

| 风险项 | 可能性 | 影响 | 应对措施 |
|--------|--------|------|----------|
| WebSocket连接不稳定 | 中 | 高 | 指数退避重连、心跳检测、消息重发 |
| 消息乱序或丢失 | 中 | 高 | 序列号排序、ACK确认、持久化存储 |
| 缓存数据不一致 | 中 | 中 | 明确过期策略、写时失效、最终一致性 |
| 双端兼容性问题 | 中 | 中 | 早期进行双端测试 |
| 高并发性能瓶颈 | 低 | 高 | Redis缓存、分页查询、Sentinel限流 |
| 图片存储压力 | 低 | 中 | 图片压缩、OSS存储、缩略图 |

### 5.2 进度风险

| 风险项 | 可能性 | 影响 | 应对措施 |
|--------|--------|------|----------|
| 人员变动 | 低 | 高 | 代码文档规范化、知识共享 |
| 需求变更 | 中 | 中 | 需求冻结，变更走流程 |
| 进度延期 | 中 | 高 | 每周进度检查，及时调整 |
| 亮点功能超时 | 中 | 中 | 亮点功能独立分支，不影响核心 |
